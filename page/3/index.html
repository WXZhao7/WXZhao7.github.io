function [p] = subposition(m,n,margin,gap)
% Uniform m*n subplot positions
% Input:
%       m,n     m*n grids
%       margin  [mL,mR,mU,mB]     
%       gap     [gW,gH]
%       
% Output:
%       p       m*n cells, the position of each grid
% 
    mL = margin(1);
    mR = margin(2);
    mU = margin(3);
    mB = margin(4);
    gW = gap(1);
    gH = gap(2);
    p = cell(m,n);
    W = (1-mL-mR-gW*(n-1))/n;
    H = (1-mU-mB-gH*(m-1))/m;
    if W<=0 | H<=0
        error('Input error!')
    end
    for im = 1:m
        for in = 1:n
            pL = mL+(W+gW)*(in-1);
            pB = mU+(H+gH)*(m-im);
            p(im,in) = {[pL,pB,W,H]};
        end
    end
end
